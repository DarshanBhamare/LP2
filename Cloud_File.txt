public class LinearSearchDemo {
    public static void search(){
        Integer[]arr=new Integer[]{1,2,5,6,7};
        Boolean found=false;
        Integer key=5;
        for(Integer i=0;i<arr.size();i++){
            if(arr[i]==key){
                System.debug('Element found at position at:'+i);
                found=true;
            }
        }
        if(found==false){
            System.debug('element not found');
        }
    }
}
